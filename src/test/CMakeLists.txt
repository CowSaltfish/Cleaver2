PROJECT(function-test)

CMAKE_MINIMUM_REQUIRED(VERSION 2.6)

SET( CMAKE_MODULE_PATH "${CMAKE_SOURCE_DIR}/")

FIND_PACKAGE(PythonLibs REQUIRED)

# Add library incl1udes
INCLUDE_DIRECTORIES(${CMAKE_BINARY_DIR}/include ${PYTHON_INCLUDE_DIRS})

# Add Source Files
FILE(GLOB srcs *.cpp)
FILE(GLOB hdrs *.h  )

MESSAGE(STATUS "PYTHON_LIBRARIES = ${PYTHON_LIBRARY}")

ADD_EXECUTABLE(function-test ${srcs} ${hdrs})
TARGET_LINK_LIBRARIES(function-test cleaver ${PYTHON_LIBRARY} )
